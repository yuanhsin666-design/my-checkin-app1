/** * Telegram 機器人後端 (Vercel Serverless 版) * 路徑必須為 /api/bot.js */ const TelegramBot = require('node-telegram-bot-api'); // 透過 Vercel 環境變數讀取 Token const token = process.env.TELEGRAM_BOT_TOKEN; const bot = new TelegramBot(token); // 模擬今日數據統計 (若不使用資料庫，此處為展示用固定數據) const getStats = () => { return { leave: 1, late: 0, normal: 25 }; }; export default async (request, response) => { try { const { body } = request; // 檢查是否為 Telegram 的更新訊息 if (body && body.message) { const chatId = body.message.chat.id; const text = body.message.text; if (text === '/start') { await bot.sendMessage(chatId, "你好！我是 DayDay 助手。輸入 /today 即可查看今日即時報表。"); } else if (text === '/status' || text === '/today') { const stats = getStats(); const report = `📊 今日報表\n---\n🏖 休假：${stats.leave}人\n⏰ 遲到：${stats.late}人\n✅ 正常出勤：${stats.normal}人`; await bot.sendMessage(chatId, report); } } // Vercel 必須回傳 200 回應給 Telegram，否則 Telegram 會重複發送訊息 response.status(200).send('OK'); } catch (error) { console.error('Error:', error); response.status(500).send('Internal Server Error'); } };
