// 這是一個 Vercel Serverless Function (後端橋樑) import { initializeApp } from 'firebase/app'; import { getFirestore, collection, getDocs } from 'firebase/firestore'; export default async function handler(req, res) { // 僅接受 Telegram 的 POST 請求 if (req.method !== 'POST') return res.status(405).send('Method Not Allowed'); // 初始化 Firebase (確保環境變數已設定) const firebaseConfig = JSON.parse(process.env.VITE_FIREBASE_CONFIG); const app = initializeApp(firebaseConfig); const db = getFirestore(app); const appId = "kawaii-unified-checkin-v4"; const { message } = req.body; if (!message || !message.text) return res.status(200).send('ok'); const text = message.text.toLowerCase(); // 處理 /today 或 /status 指令 if (text.includes('/today') || text.includes('/status')) { const today = new Date(); // 確保日期字串格式與前端寫入的一致 (toDateString) const todayKey = today.toDateString(); let report = `📅 今日考勤回報 (${today.getMonth()+1}/${today.getDate()})\n`; report += `━━━━━━━━━━━━━\n`; // 定義要查詢的成員名單 (與前端一致) const staffList = [ { id: 'staff_01', name: '小葵 🌻' }, { id: 'staff_02', name: '阿健 🛠️' }, { id: 'staff_03', name: '星星 🌟' }, { id: 'staff_04', name: '妮妮 🎀' }, { id: 'staff_05', name: '波波 🐶' } ]; let foundAny = false; // 遍歷資料庫抓取數據 for (const staff of staffList) { try { const colRef = collection(db, 'artifacts', appId, 'public', 'data', `records_${staff.id}`); const snapshot = await getDocs(colRef); // 在該成員的集合中尋找今天的紀錄 const todayRec = snapshot.docs.find(doc => doc.data().dayKey === todayKey); if (todayRec) { const data = todayRec.data(); const statusEmoji = data.label === '上' ? '✅' : (data.label === '休' ? '💤' : '📝'); report += `${statusEmoji} ${staff.name}：${data.label}\n`; foundAny = true; } else { report += `⚪ ${staff.name}：(尚未打卡)\n`; } } catch (err) { console.error(`Error fetching ${staff.id}:`, err); } } if (!foundAny) { report = `📭 今日尚未有任何打卡紀錄喔！`; } else { report += `━━━━━━━━━━━━━\n🤖 系統自動查詢完成`; } // 回傳 JSON 給 Telegram Webhook 自動發送 return res.status(200).json({ method: 'sendMessage', chat_id: message.chat.id, text: report }); } // 其他訊息不處理 res.status(200).send('ok'); }
